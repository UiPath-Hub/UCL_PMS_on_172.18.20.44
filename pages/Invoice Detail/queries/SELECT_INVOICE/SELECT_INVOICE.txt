SELECT TOP 1 invoice.*,
pdf.PDF_INVOICE_DRAFT_FILE as FILE_DRAFT,
pdf.PDF_INVOICE_FILE as FILE_ORIGINAL 
FROM dbo.PMS_INVOICE_QAS invoice
LEFT JOIN dbo.PMS_INVOICE_FILE pdf on invoice.INVOICE_ID = pdf.INVOICE_ID
WHERE 
({{Configs.INVOICE_ID == "INVOICE_NO"?1:0}} = 0 or
{{appsmith.store.INIT[Configs.INVOICE_ID]}} = invoice.INVOICE_NO) AND
({{Configs.INVOICE_ID == "INVOICE_ID"?1:0}} = 0 or
{{appsmith.store.INIT[Configs.INVOICE_ID]}} = invoice.INVOICE_ID)

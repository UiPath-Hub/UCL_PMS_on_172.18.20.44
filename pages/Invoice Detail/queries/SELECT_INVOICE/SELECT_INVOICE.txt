SELECT TOP 1 dbo.PMS_INVOICE_LM.*,
pdf.PDF_INVOICE_DRAFT_FILE as FILE_DRAFT,
pdf.PDF_INVOICE_FILE as FILE_ORIGINAL 
FROM dbo.PMS_INVOICE_LM
LEFT JOIN dbo.PMS_INVOICE_FILE pdf on dbo.PMS_INVOICE_LM.INVOICE_ID = pdf.INVOICE_ID
WHERE 
({{Configs.INVOICE_ID == "INVOICE_NO"?1:0}} = 0 or
{{appsmith.store.INIT[Configs.INVOICE_ID]}} = dbo.PMS_INVOICE_LM.INVOICE_NO) AND
({{Configs.INVOICE_ID == "INVOICE_ID"?1:0}} = 0 or
{{appsmith.store.INIT[Configs.INVOICE_ID]}} = dbo.PMS_INVOICE_LM.INVOICE_ID)

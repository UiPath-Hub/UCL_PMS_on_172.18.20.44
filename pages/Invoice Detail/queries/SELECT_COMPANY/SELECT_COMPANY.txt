SELECT 
	COM.COMPANY_ID,
	case when COM.COMPANY_LANGUAGE_TYPE='EN' then  COM.[COMPANY_NAME_EN] else COM.[COMPANY_NAME_TH] end as COMPANY_NAME,
	
	case COM.OVERWRITE_BILLING_ADDRESS when 1 then BILL.BILLING_COMPANY_ADD_NO else COM.COMPANY_ADD_NO end,
	case COM.OVERWRITE_BILLING_ADDRESS when 1 
		then case when COM.COMPANY_LANGUAGE_TYPE='EN' then BILL.BILLING_COMPANY_ROAD_EN else BILL.BILLING_COMPANY_ROAD_TH end
		else case when COM.COMPANY_LANGUAGE_TYPE='EN' then COM.COMPANY_ROAD_EN else COM.COMPANY_ROAD_TH end end as COMPANY_ROAD,
	case COM.OVERWRITE_BILLING_ADDRESS when 1 
		then case when COM.COMPANY_LANGUAGE_TYPE='EN' then BILL.BILLING_COMPANY_DISTRICT_EN else BILL.BILLING_COMPANY_DISTRICT_TH end
		else case when COM.COMPANY_LANGUAGE_TYPE='EN' then COM.COMPANY_DISTRICT_EN else COM.COMPANY_DISTRICT_TH end end as COMPANY_DISTRICT,
	case COM.OVERWRITE_BILLING_ADDRESS when 1 then BILL.BILLING_COMPANY_POSTAL_CODE else COM.COMPANY_POSTAL_CODE end as COMPANY_POSTAL_CODE,
	case COM.OVERWRITE_BILLING_ADDRESS when 1 then BILL.BILLING_COMPANY_FLOOR else COM.COMPANY_FLOOR end as COMPANY_FLOOR ,
	case COM.OVERWRITE_BILLING_ADDRESS when 1 then BILL.BILLING_COMPANY_MOO else COM.COMPANY_MOO end as COMPANY_MOO ,
	case COM.OVERWRITE_BILLING_ADDRESS when 1
		then case when COM.COMPANY_LANGUAGE_TYPE='EN' then BILL.BILLING_COMPANY_SOI_EN else BILL.BILLING_COMPANY_SOI_TH end
		else case when COM.COMPANY_LANGUAGE_TYPE='EN' then COM.COMPANY_SOI_EN else COM.COMPANY_SOI_TH end   end as COMPANY_SOI ,
	case COM.OVERWRITE_BILLING_ADDRESS when 1
		then case when COM.COMPANY_LANGUAGE_TYPE='EN' then BILL.BILLING_COMPANY_SUB_DISTRICT_EN else BILL.BILLING_COMPANY_SUB_DISTRICT_TH end 
		else case when COM.COMPANY_LANGUAGE_TYPE='EN' then COM.COMPANY_SUB_DISTRICT_EN else COM.COMPANY_SUB_DISTRICT_TH end   end as COMPANY_SUB_DISTRICT ,
	case COM.OVERWRITE_BILLING_ADDRESS when 1 
		then case when COM.COMPANY_LANGUAGE_TYPE='EN' then BILL.BILLING_COMPANY_PROVINCE_EN else BILL.BILLING_COMPANY_PROVINCE_TH end 
		else case when COM.COMPANY_LANGUAGE_TYPE='EN' then COM.COMPANY_PROVINCE_EN else COM.COMPANY_PROVINCE_TH end   end as COMPANY_PROVINCE ,
	case COM.OVERWRITE_BILLING_ADDRESS when 1 
		then BILL.BILLING_TAX_ID
		else COM.TAX_ID end as TAX_ID,
  
	case CONTACT.COMPANY_CONTACT_LANGUAGE_TYPE when 'EN' then CONTACT.COMPANY_CONTACT_FIRST_NAME_EN+' '+CONTACT.COMPANY_CONTACT_SUR_NAME_EN else
	CONTACT.COMPANY_CONTACT_FIRST_NAME_TH+' '+CONTACT.COMPANY_CONTACT_SUR_NAME_TH end AS CONTACT_NAME,
	CONTACT.COMPANY_CONTACT_TELEPHONE,
	CONTACT.COMPANY_CONTACT_EMAIL 
from [dbo].[PMS_COMPANY_LM] AS COM
LEFT JOIN dbo.PMS_CONTACT_PERSON_DETAIL_LM contactPair on contactPair.PAIRED_CONTACT_ID = COM.PRIORITY_CONTACT AND contactPair.PAIRED_COMPANY_ID = COM.COMPANY_ID AND COM.PRIORITY_CONTACT IS NOT NULL
LEFT JOIN dbo.PMS_COMPANY_CONTACT_LM AS CONTACT ON contactPair.PAIRED_CONTACT_ID = CONTACT.COMPANY_CONTACT_ID AND CONTACT.DELETE_FLAG = 0
LEFT JOIN dbo.PMS_COMPANY_BILLING_INFORMATION AS BILL ON COM.COMPANY_ID = BILL.COMPANY_ID AND BILL.DELETE_FLAG = 0
WHERE  COM.COMPANY_ID = {{this.params.ID}}

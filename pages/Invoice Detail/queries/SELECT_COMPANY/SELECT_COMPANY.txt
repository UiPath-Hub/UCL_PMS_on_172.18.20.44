SELECT 
COM.COMPANY_ID,
case when COM.COMPANY_LANGUAGE_TYPE='TH' then COM.[COMPANY_NAME_TH] else COM.[COMPANY_NAME_EN] end as COMPANY_NAME,
COM.COMPANY_ADD_NO,
	case when COM.COMPANY_LANGUAGE_TYPE='TH' then COM.COMPANY_ROAD_TH else COM.COMPANY_ROAD_EN end as COMPANY_ROAD,
	case when COM.COMPANY_LANGUAGE_TYPE='TH' then COM.COMPANY_DISTRICT_TH else COM.COMPANY_DISTRICT_EN end as COMPANY_DISTRICT,
	COM.COMPANY_POSTAL_CODE,
	COM.COMPANY_FLOOR,
	COM.COMPANY_MOO,
	case when COM.COMPANY_LANGUAGE_TYPE='TH' then COM.COMPANY_SOI_TH else COM.COMPANY_SOI_EN end as COMPANY_SOI,
	case when COM.COMPANY_LANGUAGE_TYPE='TH' then COM.COMPANY_SUB_DISTRICT_TH else COM.COMPANY_SUB_DISTRICT_EN end as COMPANY_SUB_DISTRICT,
	case when COM.COMPANY_LANGUAGE_TYPE='TH' then COM.COMPANY_PROVINCE_TH else COM.COMPANY_PROVINCE_EN end as COMPANY_PROVINCE,
	COM.TAX_ID,
CONTACT.CONTACT_NAME,
CONTACT.COMPANY_CONTACT_TELEPHONE,
CONTACT.COMPANY_CONTACT_EMAIL 
from [dbo].[PMS_COMPANY_LM] AS COM
LEFT JOIN (
    SELECT 
        assigned.PAIRED_COMPANY_ID,
        case when CON.COMPANY_CONTACT_LANGUAGE_TYPE='TH' then CON.COMPANY_CONTACT_FIRST_NAME_TH + ' ' + CON.COMPANY_CONTACT_SUR_NAME_TH 
		else CON.COMPANY_CONTACT_FIRST_NAME_EN + ' ' + CON.COMPANY_CONTACT_SUR_NAME_EN end AS CONTACT_NAME,
        CON.COMPANY_CONTACT_TELEPHONE,
				CON.COMPANY_CONTACT_EMAIL,
        assigned.COMPANY_CONTACT_ROLE,
        ROW_NUMBER() OVER (PARTITION BY assigned.PAIRED_COMPANY_ID ORDER BY assigned.CREATE_DATE ASC) AS rn
    FROM PMS_CONTACT_PERSON_DETAIL_LM assigned
    LEFT JOIN PMS_COMPANY_CONTACT_LM CON ON CON.COMPANY_CONTACT_ID = assigned.PAIRED_CONTACT_ID
    WHERE assigned.COMPANY_CONTACT_ROLE = N'ผู้ประสานงาน' AND CON.DELETE_FLAG = 0
) AS CONTACT ON COM.COMPANY_ID = CONTACT.PAIRED_COMPANY_ID AND CONTACT.rn = 1
WHERE  COM.COMPANY_ID = {{this.params.ID}}

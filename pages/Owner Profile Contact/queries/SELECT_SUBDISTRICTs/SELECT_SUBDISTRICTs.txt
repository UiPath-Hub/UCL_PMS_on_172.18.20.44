DECLARE @OWNER_CONTACT_PROVINCE_TH NVARCHAR(MAX) = {{this.params.OWNER_CONTACT_PROVINCE_TH}},
@OWNER_CONTACT_DISTRICT_TH NVARCHAR(MAX) = {{this.params.OWNER_CONTACT_DISTRICT_TH}};

IF (SELECT COUNT(*) FROM dbo.TH_ADDRESSES WHERE PROVINCE = @OWNER_CONTACT_PROVINCE_TH AND DISTRICT = @OWNER_CONTACT_DISTRICT_TH)!=0

BEGIN
SELECT DISTINCT SUBDISTRICT FROM dbo.TH_ADDRESSES WHERE PROVINCE = @OWNER_CONTACT_PROVINCE_TH AND DISTRICT = @OWNER_CONTACT_DISTRICT_TH;
END
ELSE SELECT '' AS SUBDISTRICT

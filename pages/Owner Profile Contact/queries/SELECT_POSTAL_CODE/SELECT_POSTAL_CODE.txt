DECLARE @OWNER_CONTACT_PROVINCE_TH NVARCHAR(MAX) = {{this.params.OWNER_CONTACT_PROVINCE_TH}},
@OWNER_CONTACT_DISTRICT_TH NVARCHAR(MAX) = {{this.params.OWNER_CONTACT_DISTRICT_TH}},
@OWNER_CONTACT_SUB_DISTRICT_TH NVARCHAR(MAX) = {{this.params.OWNER_CONTACT_SUB_DISTRICT_TH}};

SELECT DISTINCT POSTAL_CODE FROM dbo.TH_ADDRESSES WHERE PROVINCE =@OWNER_CONTACT_PROVINCE_TH  AND DISTRICT =@OWNER_CONTACT_DISTRICT_TH  AND SUBDISTRICT = @OWNER_CONTACT_SUB_DISTRICT_TH

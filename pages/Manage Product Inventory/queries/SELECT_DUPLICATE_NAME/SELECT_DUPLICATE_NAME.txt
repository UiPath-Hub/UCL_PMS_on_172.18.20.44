DECLARE @USER_TOKEN nvarchar(50) = {{appsmith.store[Configs.userSession].TOKEN}}
DECLARE @PAIRED_OWNER_ID dbo.NEW_ID = [dbo].fn_getPAIRED_OWNER_IDByToken(@USER_TOKEN)
DECLARE @FREE_SPACE NVARCHAR(10) = COALESCE((SELECT TOP(1) FREE_SPACE FROM SQL_GLOBAL_VARIABLES), 'FREE_SPACE')

if @FREE_SPACE = {{INVENTORY_NAME.text}} 
SELECT 2 as COUNT
else
BEGIN
SELECT COUNT(*) AS COUNT  FROM dbo.PMS_PRODUCT_INVENTORY_LM WHERE INVENTORY_NAME = {{INVENTORY_NAME.text}} AND DELETE_FLAG = 0 AND OWNER_ID = @PAIRED_OWNER_ID
end

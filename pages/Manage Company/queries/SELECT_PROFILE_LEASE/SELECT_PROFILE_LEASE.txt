SELECT profile.* FROM dbo.PMS_COMPANY_PROFILE_LM as profile
LEFT JOIN dbo.PMS_COMPANY_PROFILE_DETAIL_LM as detail ON profile.COMPANY_PROFILE_ID = detail.COMPANY_PROFILE_ID
LEFT JOIN dbo.PMS_FORMULA_LM as formula ON formula.FORMULA_ID = profile.FORMULA_ID
WHERE profile.COMPANY_ID = {{COMPANY_ID.text}}

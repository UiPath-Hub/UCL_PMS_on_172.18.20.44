--update temp need @COMPANY_ID = "TEMP" , @USER_TOKEN , @NEWBRANCH

DECLARE @ResultCode NVARCHAR(10) ,
    @ResultMessages NVARCHAR(MAX);
execute dbo.[UPDATE_REMARK]
			@BILLING_REMARK_ID  = {{this.params.BILLING_REMARK_ID}}
		,@COMPANY_ID = {{COMPANY_ID.text||"TEMP"}}
		,@BILLING_REMARK_TYPE  = {{this.params.BILLING_REMARK_TYPE}}
		,@INVOICE_REMARK_TYPE= {{this.params.INVOICE_REMARK_TYPE}}
		,@BILLING_REMARK_PRODUCT_TYPE_ID= {{this.params.BILLING_REMARK_PRODUCT_TYPE_ID}}
		,@BILLING_REMARK_PRODUCT_TYPE_NAME= {{this.params.BILLING_REMARK_PRODUCT_TYPE_NAME}}
		,@BILLING_REMARK_GROUP_ID= {{this.params.BILLING_REMARK_GROUP_ID}}
		,@BILLING_REMARK_DETAIL= {{this.params.BILLING_REMARK_DETAIL}}
		,@USER_TOKEN = {{appsmith.store[Configs.userSession].TOKEN}}
		,@NEWBRANCH = {{appsmith.store.NEWBRANCH}}
		,@IS_UPDATE = 1
		,@ResultCode=@ResultCode OUTPUT
		,@ResultMessages=@ResultMessages OUTPUT
		
	SELECT @ResultCode AS [RESULT_CODE],
       @ResultMessages AS [RESULT_MESSAGES]

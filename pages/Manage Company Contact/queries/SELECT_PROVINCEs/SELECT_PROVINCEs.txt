SELECT 
    (SELECT TOP(1) ID FROM dbo.TH_ADDRESSES tt WHERE DistinctProvinces.PROVINCE_TH = tt.PROVINCE) as ID,
    PROVINCE_TH,
		PROVINCE_EN
FROM (
    SELECT DISTINCT t.PROVINCE as PROVINCE_TH,e.PROVINCE as PROVINCE_EN
    FROM dbo.TH_ADDRESSES t join dbo.EN_ADDRESSES e on t.ID = e.ID 
) AS DistinctProvinces
ORDER BY PROVINCE_TH ASC;
